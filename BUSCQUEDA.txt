DECLARE   
   CURSOR C1 IS SELECT  NOMBRE_PRODUCTO FROM PRODUCTO WHERE ID_CATEGORIA='2';
  lNOMBRE_PRODUCTO PRODUCTO.NOMBRE_PRODUCTO %TYPE; 
   
BEGIN
   OPEN C1;
   LOOP 
      FETCH C1 INTO lNOMBRE_PRODUCTO;
      EXIT WHEN C1%NOTFOUND;
      DBMS_OUTPUT.PUT_LINE(lNOMBRE_PRODUCTO);
   END LOOP;
   CLOSE C1;
END;
